public void PrintStatistics(double[] array, int count)
{
	double maxElement = double.MinValue;

	for (int i = 0; i < count; i++)
	{
		if (array[i] > maxElement)
		{
			maxElement = array[i];
		}
	}

	PrintMaximalValueInArray(maxElement);

	double minElement = double.MaxValue;

	for (int i = 0; i < count; i++)
	{
		if (array[i] < minElement)
		{
			minElement = array[i];
		}
	}

	PrintMinimalValueInArray(minElement);

	double sumOfAllElements = 0;

	for (int i = 0; i < count; i++)
	{
		sumOfAllElements += array[i];
	}

	PrintAverageValueOfArray(sumOfAllElements / count);
}
